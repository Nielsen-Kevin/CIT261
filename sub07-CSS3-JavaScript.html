Style color, Style top

The message in the game changes color basE on if your guess is wrong or right. The message also looked to static so I made it move to give it some life.

function message(status) { // Setup messages from array random_number = Math.round(Math.random() * (messages[status].length - 1)); document.getElementById('message').innerHTML = messages[status][random_number]; document.getElementById('message').style.color = txtColor[status]; // Gives the message some life let pos = -20; let id = setInterval(frame, 5); function frame() { if (pos == 0) { clearInterval(id); } else { pos++; document.getElementById("message").style.top = pos + "px"; } } }
