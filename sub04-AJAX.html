Using XMLHTTPRequest to Consume a JSON Web Service

Assignment Notes: This is the prototype API request page build to show holidays throughout the year. The Holiday API is coded in PHP to calculate the correct dates for a list of holidays and output them in a JSON document that can be requested and read by the AJAX script. I add this to my calendar application also. It started just for Javasricpt Objects assignment and grow from there.

let month = parseInt(document.getElementById('m').value); let year = parseInt(document.getElementById('y').value); let qStr = ''; if(!isNaN(year) && year >= 1970 && year <= 2037) { if(!isNaN(month) && month >= 1 && month <= 12) { qStr = '?y=' + year + '&m=' + month; } else { qStr = '?y=' + year; } } else { alert('Year is invalid!'); } var xmlhttp = new XMLHttpRequest(); xmlhttp.onreadystatechange = function() { if (this.readyState == 4 && this.status == 200) { var data = JSON.parse(this.responseText); var html = '<div>Holiday</div>'; html += '<div>Date</div>'; html += '<div>Day of Week</div>'; html += '<div>Timestamp</div>'; for(let key in data) { html += '<div>' + data[key].name + '</div>'; html += '<div>' + data[key].month + '/' + data[key].day + '/' + data[key].year + '</div>'; html += '<div>' + data[key].weekday + '</div>'; html += '<div>' + data[key].time + '</div>'; } document.getElementById("display").innerHTML = html; } }; xmlhttp.open('GET', 'http://forgeofatlas.com/cit261/holidayAPI.php' + qStr , true); xmlhttp.send(); }
